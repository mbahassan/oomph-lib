37 /// Constructor: Pass the filename of the hexahedral file
38 /// The assumptions are that the nodes have been assigned boundary
39 /// information which is used in the nodal construction to make sure
40 /// that BoundaryNodes are constructed. Any additional boundaries are
41 /// determined from the face boundary information.
42 //======================================================================
43 GmshScaffoldMesh::GmshScaffoldMesh(const std::string& file_name, bool verbose)
44 {
45 // Process the element file
46 // --------------------------
47 GMSH::Gmsh rg(file_name, verbose);
48 boundaries = rg.boundaries;
49 orderedBC = rg.orderedBC;
50 std::vector<GMSH::Node> nodes = rg.getNodes();
51 std::vector<GMSH::Edge> edges = rg.getEdges();
52
53 // dimension of the mesh
54 const unsigned dim = rg.getMeshDim();
55 std::vector<GMSH::Quad> faces = rg.getQuads();
56 std::vector<GMSH::Hexa> bricks= rg.getHexas();
57
58
59 unsigned n_local_face = rg.getFacesPerElement();
60 unsigned n_local_edge = rg.getEdgesPerElement();
61
62
63 if (dim ==2){
64 std::cout<< "Mesh is 2D!" <<std::endl;
65 exit(-1);
66 }
67 // Read in number of elements
68 unsigned n_element = bricks.size();
69
70 // Read in number of nodes per element // this should be constant = 8
71 unsigned n_local_node = rg.getNodesPerElement();
72
73
74 // Throw an error if we have anything other than [Hexas or Quads] but linear simplices
75 if (n_local_node != Hexaheadral::nNodes && n_local_node != 4)
76 {
77 std::ostringstream error_stream;
78 error_stream
79 << "Gmsh Reader should only be used to generate 8-noded hexahedra!\n"
80 << "Your gmsh input file, contains data for " << n_local_node
81 << "-noded hexahedra" << std::endl;
82
83 throw OomphLibError(
84 error_stream.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
85 }
86
87 // Element attributes may be used to distinguish internal regions
88 // NOTE: This stores doubles because gmsh forces us to!
89 // Element_attribute.resize(n_element, 0.0);
90
91
92 // Resize storage for the global node numbers listed element-by-element
93 Global_node.resize(n_element*n_local_node);
94
95 // Initialise (global) node counter
96 unsigned k = 0;
97
98 // Read in global node numbers
99 for (auto &brick: bricks)
100 {
101 for (auto node: brick.nodesTag) {
102 Global_node[k] = node;
103 k++;
104 }
105 }
106
107
108 // Resize the Element vector
109 Element_pt.resize(n_element);
110
111 // Process node file
112 //--------------------
113
114 // Read in the number of nodes
115 unsigned n_node = nodes.size();
116
117 // Create a vector of boolean so as not to create the same node twice
118 std::vector<bool> done(n_node, false);
119
120 // Resize the Node vector
121 Node_pt.resize(n_node);
122
123
124 // Create storage for nodal positions and boundary markers
125 Vector<double> x_node(n_node);
126 Vector<double> y_node(n_node);
127 Vector<double> z_node(n_node);
128 Vector<Vector<int>> bound(n_node);
129
130 // Check if there are attributes
131 // If not
132 for (unsigned i = 0; i < n_node; i++)
133 {
134 x_node[i] = nodes[i].coord[0];
135 y_node[i] = nodes[i].coord[1];
136 z_node[i] = nodes[i].coord[2];
137
138 bound[i].resize(nodes[i].boundaries.size());
139 int c = 0;
140 for (auto it = nodes[i].boundaries.begin(); it != nodes[i].boundaries.end(); ++it) {
141 bound[i][c] = *it;
142 c++;
143 }
144
145 }
146
147 // Determine the highest boundary index
148 //------------------------------------
149 unsigned n_bound = rg.getBoundSize();
150
151 // Process face to extract boundary faces
152 //--------------------------------------------
153
154 // Number of faces in faces
155 unsigned n_face = faces.size();
156
157
158 // Storage for the global node numbers (in the gmsh 1-based
159 // numbering scheme!) of the first, second and third node in
160 // each segment
161 Vector<unsigned> first_node(n_face);
162 Vector<unsigned> second_node(n_face);
163 Vector<unsigned> third_node(n_face);
164 Vector<unsigned> forth_node(n_face);
165
166 // Storage for the boundary marker for each face
167 Vector<int> face_boundary(n_face);
168
169 // Dummy for global face number
170 unsigned dummy_face_number;
171
172 // Storage for the (boundary) faces associated with each node.
173 // Nodes are indexed using Reader's 1-based scheme, which is why
174 // there is a +1 here
175 Vector<std::set<unsigned>> node_on_faces(n_node);
176
177 for (int i = 0; i < faces.size(); ++i)
178 {
179 first_node[i] = faces[i].nodesTag[0];
180 second_node[i] = faces[i].nodesTag[1];
181 third_node[i] = faces[i].nodesTag[2];
182 forth_node[i] = faces[i].nodesTag[3];
183
184 // \todo: what if more than b.c.?
185 auto it = faces[i].boundaries.begin();
186 face_boundary[i] = *it;
187
188 // Add the face index to each node
189 node_on_faces[first_node[i]].insert(i);
190 node_on_faces[second_node[i]].insert(i);
191 node_on_faces[third_node[i]].insert(i);
192 node_on_faces[forth_node[i]].insert(i);
193 }
194
195 // Set number of boundaries
196 if (n_bound > 0)
197 {
198 this->set_nboundary(n_bound);
199 }
200
201 // Translate enumeration
202 Vector<unsigned> gmsh_to_oomph_node(8);
203 gmsh_to_oomph_node[0] = 0;
204 gmsh_to_oomph_node[1] = 1;
205 gmsh_to_oomph_node[2] = 3;
206 gmsh_to_oomph_node[3] = 2;
207
208 gmsh_to_oomph_node[4] = 4;
209 gmsh_to_oomph_node[5] = 5;
210 gmsh_to_oomph_node[6] = 7;
211 gmsh_to_oomph_node[7] = 6;
212
213 // Create the elements
214 unsigned counter = 0;
215 for (unsigned e = 0; e < n_element; e++)
216 {
217 Element_pt[e] = new QElement<3, 2>;
218
219 // Loop over the other nodes
220 for (unsigned j = 0; j < n_local_node; j++)
221 {
222 unsigned global_node_number = Global_node[counter];
223 if (!done[global_node_number])
224 {
225 // If we're on a boundary
226 // [gmsh convention 1 or greater is a boundary, 0 not a boundary]
227 for (int i = 0; i < bound[global_node_number].size(); ++i)
228 {
229 if (bound[global_node_number][i] > 0)
230 {
231 // Construct the boundary node
232 Node_pt[global_node_number] = finite_element_pt(e)->construct_boundary_node(gmsh_to_oomph_node[j]);
233
234 // Add to the boundary lookup scheme.
235 // bound[global_node_number]-1 because vector indexing starts from 0
236 add_boundary_node(bound[global_node_number][i] - 1 ,Node_pt[global_node_number]);
237 }
238 else
239 {
240 Node_pt[global_node_number] = finite_element_pt(e)->construct_node(gmsh_to_oomph_node[j]);
241 }
242
243 done[global_node_number] = true;
244 Node_pt[global_node_number]->x(0) = x_node[global_node_number];
245 Node_pt[global_node_number]->x(1) = y_node[global_node_number];
246 Node_pt[global_node_number]->x(2) = z_node[global_node_number];
247 }
248
249 }
250 else
251 {
252 // Otherwise copy the pointer over
253 finite_element_pt(e)->node_pt(gmsh_to_oomph_node[j]) = Node_pt[global_node_number];
254 }
255 counter++;
256 }
257 }
258
259
260 // Resize the "matrix" that stores the boundary id for each
261 // face in each element.
262 Face_boundary.resize(n_element);
263 Face_index.resize(n_element);
264 Edge_index.resize(n_element);
265
266
267 // 0-based index scheme used to construct a global lookup for
268 // each face that will be used to uniquely construct interior
269 // face nodes.
270 // The faces that lie on boundaries will have already been allocated so
271 // we can start from there
272 for (auto& face: faces) {
273 if (*face.boundaries.begin() == 0) Nglobal_face++;
274 }
275
276
277 // Storage for the edges associated with each node.
278 // Nodes are indexed using Reader's 1-based scheme, which is why
279 // there is a +1 here
280 Vector<std::set<unsigned>> node_on_edges(n_node);
281
282 // 0-based index scheme used to construct a global lookup for each
283 // edge that will be used to uniquely construct interior edge nodes
284 Nglobal_edge = edges.size();
285
286 // Conversion from the edge numbers to the nodes at the end
287 // of each edge
288
289 // Loop over the elements
290 for (unsigned e = 0; e < bricks.size(); e++)
291 {
292 // Each element has six faces
293 Face_boundary[e].resize(n_local_face);
294 Face_index[e].resize(n_local_face);
295 // By default, each face is NOT on a boundary
296 for (unsigned i = 0; i < n_local_face; i++)
297 {
298 Face_boundary[e][i] = 0;
299 }
300
301 // Read out the global node numbers of the nodes from
302 // tetgen's 1-based numbering.
303 // The offset is to match the offset used above
304 const unsigned element_offset = e * n_local_node;
305 unsigned glob_num[8] = {0, 0, 0, 0, 0, 0, 0, 0};
306 for (unsigned i = 0; i < n_local_node; ++i)
307 {
308 glob_num[i] = Global_node[element_offset + i];
309 }
310
311 // Now we know the global node numbers of the elements' four nodes
312 // in tetgen's 1-based numbering.
313
314 // Determine whether any of the element faces have already been allocated
315 // an index. This may be because they are located on boundaries, or have
316 // already been visited The global index for the i-th face will be stored
317 // in Face_index[i]
318
319 GMSH::Hexa& brick = bricks[e];
320 // Loop over the local faces in the element
321 for (unsigned i = 0; i < brick.quadsTag.size(); ++i)
322 {
323 //std::vector<int> facesTags = brick.quadsTag;
324
325 // Otherwise we already have a face
326 GMSH::Quad& face = faces[brick.quadsTag[i]];
327
328 auto it_b = face.boundaries.begin();
329 if (*it_b > 0 )
330 {
331 // Set the face index
332 Face_index[e][i] = face.quadTag;
333
334 // Allocate the boundary index, if it's a boundary
335 Face_boundary[e][i] = *it_b;
336
337 // Add the nodes to the boundary look-up scheme in
338 // oomph-lib (0-based) index
339 for (int nodeTag : face.nodesTag)
340 {
341 auto b = *it_b - 1;
342 // Don't add the omitted node
343 add_boundary_node(b, Node_pt[nodeTag]);
344 }
345 }
346 else
347 {
348 // Allocate the next global index
349 Face_index[e][i] = face.quadTag;
350
351 }
352 } // end of loop over the faces
353
354
355 Edge_index[e].resize(n_local_edge);
356
357 // Loop over the element edges and assign global edge numbers // we have 12 edges on 1 hexa element
358 for (unsigned i = 0; i < brick.edgesTag.size(); ++i)
359 {
360 std::vector<int> edgesTags = brick.edgesTag;
361
362 GMSH::Edge& edge = edges[edgesTags[i]];
363
364 // Allocate the next global index
365 Edge_index[e][i] = edge.edgeTag;
366
367 // Otherwise we already have an edge
368
369 // Set the edge index from the result of the intersection
370 Edge_index[e][i] = edge.edgeTag;
371
372 } // end for edge
373
374 } // end for e
375
376 // Now determine whether any edges lie on boundaries by using the
377 // face boundary scheme and
378
379 // Resize the storage
380 Edge_boundary.resize(Nglobal_edge, false);
381
382 // Now loop over all the boundary faces and mark that all edges
383 // must also lie on the boundary
384 for (unsigned i = 0; i < edges.size(); ++i)
385 {
386 if (*edges[i].boundaries.begin() > 0)
387 {
388 Edge_boundary[i] = true;
389 }
390 }
391 } // end of constructor
