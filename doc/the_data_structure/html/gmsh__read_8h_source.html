231 template<typename T>
232 std::vector<T> operator^(const std::vector<T>& lhs, const std::vector<T>& rhs)
233 {
234 if (lhs.size() != rhs.size()) {
235 throw std::invalid_argument("Vectors must have the same dimensionality for cross product.");
236 }
237
238 std::vector<T> result(lhs.size());
239
240 result[0] = lhs[1] * rhs[2] - lhs[2] * rhs[1];
241 result[1] = lhs[2] * rhs[0] - lhs[0] * rhs[2];
242 result[2] = lhs[0] * rhs[1] - lhs[1] * rhs[0];
243
244 return result;
245 }

249 T operator&(const std::vector<T>& lhs, const std::vector<T>& rhs)
250 {
251 if (lhs.size() != rhs.size()) {
252 throw std::invalid_argument("Vectors must have the same length for dot product.");
253 }
254
255 T result = 0;
256
257 size_t n = lhs.size();
258
