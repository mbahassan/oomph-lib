36 /// Mesh that is based on input files generated by the hexahedral
37 /// mesh generator gmsh.
38 //======================================================================
39 class GmshScaffoldMesh : public virtual Mesh
40 {
41 public:
42 /// Empty constructor
43 GmshScaffoldMesh() = default;
44
45 /// Constructor: Pass the filename of the hexahedral file
46 GmshScaffoldMesh(const std::string& file_name, bool verbose);
47
48 /// Empty destructor
49 ~GmshScaffoldMesh() {}
50
51 /// Return the global node of each local node
52 /// listed element-by-element e*n_local_node + n_local
53 /// Note that the node numbers are indexed from 1
54 unsigned global_node_number(const unsigned& i)
55 {
56 return Global_node[i];
57 }
58
59 /// Return the boundary id of the i-th face in the e-th element:
60 /// This is zero-based as in gmsh. Zero means the face is not
61 /// on a boundary. Postive numbers identify the boundary.
62 /// Will be reduced by one to identify the oomph-lib boundary.
63 unsigned face_boundary(const unsigned& e, const unsigned& i) const
64 {
65 return Face_boundary[e][i];
66 }
67
68 /// Return the number of internal edges
69 unsigned nglobal_edge()
70 {
71 return Nglobal_edge;
72 }
73
74 /// Return a boolean indicating whether the i-th global
75 /// edge is on a boundary
76 bool edge_boundary(const unsigned& i)
77 {
78 return Edge_boundary[i];
79 }
80
81 /// Return the global index of the i-th edge in the e-th element:
82 /// The global index starts from zero
83 unsigned edge_index(const unsigned& e, const unsigned& i) const
84 {
85 return Edge_index[e][i];
86 }
87
88 /// Return the number of internal face
89 unsigned nglobal_face()
90 {
91 return Nglobal_face;
92 }
93
94 /// Return the global index of the i-th face in the e-th element:
95 /// The global index starts from zero
96 unsigned face_index(const unsigned& e, const unsigned& i) const
97 {
98 return Face_index[e][i];
99 }
100
101 /// Return the attribute of the element e.
102 /// NOTE: Attributes are doubles because gmsh forces us to! We only
103 /// use them for region IDs
104 double element_attribute(const unsigned& e) const
105 {
106 return Element_attribute[e];
107 }
108
109 std::vector<GMSH::Boundary> boundaries;
110
111 std::unordered_map<int, int> orderedBC;
112
113 protected:
114 /// Storage for the number of global faces
115 unsigned Nglobal_face =0;
116
117 /// Storage for the number of global edges
118 unsigned Nglobal_edge = 0;
119
120 /// Storage for global node numbers listed element-by-element
121 Vector<unsigned> Global_node ;
122
123 /// Vector of booleans to indicate whether a global edge lies
124 /// on a boundary
125 std::vector<bool> Edge_boundary ;
126
127 /// Vector of vectors containing the boundary ids of the
128 /// elements' faces
129 Vector<Vector<unsigned>> Face_boundary ;
130
131 /// Vector of vectors containing the global edge index of
132 // the elements' edges
133 Vector<Vector<unsigned>> Edge_index ;
134
135 /// Vector of vectors containing the global edge index of
136 // the elements' edges
137 Vector<Vector<unsigned>> Face_index;
138
139 /// Vector of double attributes for each element.
140 /// NOTE: This stores doubles because gmsh forces us to! We only
141 /// use it for region IDs
142 Vector<double> Element_attribute;
143 };
144
145} // namespace oomph
