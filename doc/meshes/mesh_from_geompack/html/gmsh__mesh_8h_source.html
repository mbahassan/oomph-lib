38 /// Quadrilateral mesh generator; Uses input from Reader.
39 /// See: http://members.shaw.ca/bjoe/
40 /// Currently only for four-noded quads -- extension to higher-order
41 /// quads should be trivial (see the corresponding classes for
42 /// triangular meshes).
43 //========================================================================
44 template<class ELEMENT>
45 class GmshMesh : public virtual BrickMeshBase
46 {
47 public:
48 /// Default Constructor
49 GmshMesh()
50 {
51 // Mesh can only be built with 3D Telements.
52 MeshChecker::assert_geometric_element<QElementGeometricBase, ELEMENT>(3);
53 }
54
55 /// Constructor with the input files
56 explicit GmshMesh(const std::string& filename, bool verbose,
57 TimeStepper* time_stepper_pt = &Mesh::Default_TimeStepper )
58 {
59 // Mesh can only be built with 3D Qelements.
60 MeshChecker::assert_geometric_element<QElementGeometricBase, ELEMENT>(3);
61
62
63 // Build scaffold
64 Tmp_mesh_pt = new GmshScaffoldMesh(filename, verbose);
65 boundaries = Tmp_mesh_pt->boundaries;
66 orderedBC = Tmp_mesh_pt->orderedBC;
67
68 // Convert mesh from scaffold to actual mesh
69 build_from_scaffold(time_stepper_pt);
70
71 // Kill the scaffold
72 delete Tmp_mesh_pt;
73 Tmp_mesh_pt = nullptr;
74 }
75
76 /// Empty destructor
77 ~GmshMesh() override = default;
78
79
80 std::vector<GMSH::Boundary> boundaries;
81
82 std::unordered_map<int, int> orderedBC;
83 protected:
84 /// Temporary scaffold mesh
85 GmshScaffoldMesh* Tmp_mesh_pt = nullptr;
86
87 /// Build mesh from scaffold
88 void build_from_scaffold(TimeStepper* time_stepper_pt);
89 };

111
